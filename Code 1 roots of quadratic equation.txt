PROGRAME TO FIND THE ROOTS OF QUADRATIC EQUATION AND ITS NATURE 

#include<iostream>
#include<cmath>
using namespace std;
int main()

void Roots(float a,float b,float c)
{
    float d,r1,r2;
    
    d=b*b-4*a*c;
    if (d == 0)
    {
        r1 = (-b/(2*a));
        cout<<"real and equal"<<r1;
    }
    else
    {
        if (d>0)
        {
            r1 = (-b+sqrt(d)/(2*a));
            r2 = (-b-sqrt(d)/(2*a));
            cout<<"Real and unequal"<<r1<<" "<<r2;
        }
        else
        {
            cout<<"Imaginary";
        }
    }
}
